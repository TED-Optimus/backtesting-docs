performance aggregator documentation

  • When backtesting a strategy for multiple periods, this library helps user to gather 5 performance metrics and saves to a local csv file. The metrics are:

    1. CAGR/Max drawdown

    2. Sharpe ratio

    3. SQN

    4. PROM (%)

    5. Average win/ Average loss

HOW TO USE

Example for performance aggregator function

  1# RSI + SMA_50 oversold system
  2
  3import numpy as np
  4import pandas as pd
  5
  6from optimus.backtesting.data import Universe, UniverseSelection
  7from optimus.backtesting.indicator import Indicator
  8from optimus.backtesting.performance import Performance
  9from optimus.backtesting.portfolio import Portfolio
 10from optimus.backtesting.selection import EquityModel, PositionSizing, Reason
 11
 12# Step 1. Import performance aggregator
 13from optimus.backtesting.performance_aggregator import PerformanceAggregator
 14
 15# Press the green button in the gutter to run the script.
 16
 17if __name__ == "__main__":
 18    # Desired output option
 19    desired_width = 1000
 20    pd.set_option("display.width", desired_width)
 21    pd.set_option("display.max_columns", 15)
 22
 23    indicator = Indicator()
 24    indicator.sma([50])
 25    indicator.rsi([2, 5])
 26    indicator.roc([100])
 27    indicator.average_daily_turnover([50])
 28
 29    # Initializing universe
 30    universe = Universe(selection=UniverseSelection.STOCKS, indicator=indicator)
 31    df = universe.df
 32
 33    date_2016 = pd.to_datetime("2016-01-01")
 34    date_2017 = pd.to_datetime("2017-01-01")
 35    date_2018 = pd.to_datetime("2018-01-01")
 36    date_2019 = pd.to_datetime("2019-01-01")
 37    date_2020 = pd.to_datetime("2020-01-01")
 38    date_2021 = pd.to_datetime("2021-01-01")
 39    date_2022 = pd.to_datetime("2022-01-01")
 40
 41    df_list = universe.df_multi_period(date_2016, date_2017, date_2018, date_2019, date_2020, date_2021, date_2022)
 42    df_list.insert(0, df[(df.index>date_2016) & (df.index<date_2022)])  # [Optional] generate data from year 2016 to 2022, and insert the generated "overall" period dataframe at the front
 43
 44    # Step 2. Initialize a performance aggregator
 45    result_aggregator = PerformanceAggregator()
 46
 47    """
 48    Entry Rules:
 49    1) RSI(2) < 5
 50    2) Close > SMA(50)
 51
 52    Exit Rules:
 53    1) RSI(5) > 40
 54    2) Exit if entered for more than 10 days
 55
 56    Ranking Criteria:
 57    Rank stocks based on Rate of change(100)
 58    """
 59
 60    for df in df_list:
 61
 62        # Step 3. Get the dates from first and last row for CAGR calculation
 63        first_day = df.index[0]
 64        last_day = df.index[-1]
 65
 66        # Entry signal
 67        df["buy_signal"] = np.where(
 68            (df["average_daily_turnover_50"].shift(1) >= 2500000)
 69            & (df["close"].shift(1) > 0.3)
 70            & (df["close"].shift(1) > df["SMA_50"].shift(1))
 71            & (df["RSI_2"].shift(1) < 5)
 72            & (df["ticker"] == df["ticker"].shift(1)),
 73            df["ROC_100"].shift(1),
 74            0,
 75        )
 76        df["buy_signal"] = df["buy_signal"].fillna(0)
 77
 78        # Exit signal
 79        df["sell_signal"] = np.where(
 80            (df["RSI_5"].shift(1) > 40)
 81            & (df["ticker"] == df["ticker"].shift(1)),
 82            1,
 83            0
 84        )
 85
 86        # Time stop
 87        df["buy_date"] = df.index
 88        df["sell_signal_2"] = np.where(
 89            (df["buy_signal"].shift(10) != 0) & (df["ticker"] == df["ticker"].shift(10)),
 90            df["buy_date"].shift(10).dt.date,
 91            pd.to_datetime("2000-01-01"),
 92        )
 93
 94        # Initializing portfolio
 95        portfolio = Portfolio(
 96            df,
 97            EquityModel.TOTAL_EQUITY,
 98            PositionSizing.EQUAL_UNIT,
 99            initial_capital=int(1e5),
100            size=10,
101        )
102
103        df.reset_index(inplace=True)
104        # Backtesting starts here!
105        for index, frame in df.groupby("date"):
106            days_count = pd.Timedelta(days=frame.index[0])
107            frame.set_index("ticker", inplace=True)
108
109            for ticker, info in portfolio.holding():
110                if frame.loc[ticker, "sell_signal_2"] == info["open_date"].date():  # 10 days sell signal
111                    portfolio.sell(
112                        ticker, Reason.EXIT, index, frame.loc[ticker, "open"], days_count
113                    )
114                elif frame.loc[ticker, "sell_signal"] == 1:
115                    portfolio.sell(
116                        ticker, Reason.EXIT, index, frame.loc[ticker, "open"], days_count
117                    )
118
119            # Sort remaining stocks that have buy signals by momentum (High to low)
120            frame = frame.loc[
121                        (frame["buy_signal"].values != 0) & (~frame.index.isin(portfolio.list()))
122                        ].sort_values("buy_signal", ascending=False)[: portfolio.free()]
123
124            # Retrieve portfolio allocation
125            for item in frame.index.to_list():
126                portfolio.buy(
127                    item,
128                    index,
129                    frame.loc[item, "open"],
130                    days_count
131                )
132
133        # Measuring performance
134        performance = Performance(portfolio, show=False)
135        performance.objective_function(need_print=False)
136        performance.performance_metric(need_print=False)
137        # Step 4. Add performance results to result_aggregator
138        result_aggregator.add_result(portfolio, performance, first_day, last_day)
139
140    # Step 5. Save aggregated results
141    result_aggregator.save_result("all_results.csv", need_print=True)

  • Assign the first and last row index (date) to variables for each loop

2last_day = df.index[-1]

  • Add the performance results after being generated with portfolio, performance, first_day and last_day

1result_aggregator.add_result(portfolio, performance, first_day, last_day)

  • Export the aggregated results to csv outside of loop
